294class ViseUnconvergedErrorHandler(orig_handlers.UnconvergedErrorHandler): 

295 

296 def correct(self): 

297 backup(orig_handlers.VASP_BACKUP_FILES) 

298 v = Vasprun(self.output_filename) 

299 actions = [{"file": "CONTCAR", 

300 "action": {"_file_copy": {"dest": "POSCAR"}}}] 

301 if not v.converged_electronic: 

302 # For SCAN try switching to CG for the electronic minimization 

303 if "SCAN" in v.incar.get("METAGGA", "").upper(): 

304 new_settings = {"ALGO": "All"} 

305 else: 

306 new_settings = {"ISTART": 1, 

307 "ALGO": "Normal", 

308 "NELMDL": -6, 

309 "BMIX": 0.001, 

310 "AMIX_MAG": 0.8, 

311 "BMIX_MAG": 0.001} 

312 

313 if all([v.incar.get(k, "") == val for k, val in 

314 new_settings.items()]): 

315 return {"errors": ["Unconverged"], "actions": None} 

316 

317 actions.append({"dict": "INCAR", 

318 "action": {"_set": new_settings}}) 

319 

320 # Instead of changing the IBRION, we reduce the EDIFF since we usually 

321 # use the constant EDIFF value as default. 

322 elif not v.converged_ionic: 

323 ediff = v.incar["EDIFF"] 

324 actions.append({"dict": "INCAR", 

325 "action": {"_set": {"EDIFF": ediff * 0.5}}}) 

326 actions.append({"dict": "INCAR", 

327 "action": {"_set": {"ADDGRID": True}}}) 

328 ViseVaspModder().apply_actions(actions) 

329 return {"errors": ["Unconverged"], "actions": actions}
