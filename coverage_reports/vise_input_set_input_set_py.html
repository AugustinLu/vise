41class ViseInputSet(VaspInputSet): 

42 """Vise version of VaspInputSet 

43 

44 Use the kwargs for the options that control vasp input set as the dist 

45 can be easily updated. Here, a list of options are listed as below. 

46 

47 Options: 

48 xc (Xc): 

49 Exchange-correlation (xc) defined in Xc. 

50 task (Task): 

51 Task defined in Task. 

52 sort_structure (bool): 

53 Whether to sort the elements using get_sorted_structure method 

54 of Structure class in pymatgen. 

55 standardize_structure (bool): 

56 Whether to convert the structure to a standardized primitive. 

57 kpt_mode (str): 

58 A string representing the k-point style that is used for the 

59 make_kpoints function. Now, "band", "primitive_uniform" and 

60 "manual_set" are supported. See make_kpoints docstring. 

61 kpt_density (float): 

62 K-point density in Angstrom along each direction used for 

63 determining k-point uniform mesh. 

64 kpt_shift (list): 

65 K-point shift in vasp definition. 

66 only_even (bool): 

67 Whether to ceil the numbers of kpoints to be even. 

68 band_ref_dist (float): 

69 A reference target distance between neighboring k-points in the 

70 path, in units of 1/Angstrom. The actual value will be as close 

71 as possible to this value, to have an integer number of points 

72 in each path. Note that seekpath default is 0.025. 

73 factor (int): 

74 Factor to be multiplied to the k-points along all the 

75 directions. This parameter should be distinguished from the 

76 kpt_density as it sets NKRED = factor for hybrid calculations:w 

77 symprec (float): 

78 Distance precision in Angstrom used for symmetry analysis. 

79 angle_tolerance (float): 

80 Angle precision in degrees used for symmetry analysis. 

81 charge (float): 

82 Charge state used for e.g., defect calculations. 

83 default_potcar (str / None): 

84 Default potcar name defined in potcar_set.yaml. 

85 By default, use "normal", but one needs to use GW potcars for GW 

86 calculation, which will be supported in the future. 

87 override_potcar_set (dict /None): 

88 Dict representation for overriding the potcar set, e.g., 

89 {"Mg": "Mg_pv", "O": "O_h"} 

90 vbm_cbm (list): 

91 The absolute valence band maximum (vbm) and conduction band minimum 

92 (cbm) with [vbm, cbm] format in eV, which are used for determining 

93 ISMEAR and the spectra region of the dos and absorption spectra. 

94 aexx (float): 

95 Exchange mixing parameter, namely AEXX in INCAR for hybrid 

96 functional calculations. 

97 hubbard_u (bool): 

98 Whether to add Hubbard U parameter. 

99 By default (None), it is switched on for LDA and GGA calculations, 

100 while switched off for hybrid functional and GW approximations. 

101 Default U-parameters are determined in u_parameter_set.yaml. 

102 ldauu (dict): 

103 Set when users want to modify U parameters from default. E.g., 

104 {"Ti": 4, "V": 3, "O": 5} 

105 ldaul (dict): 

106 Set when users want to modify orbitals from default. 

107 If ldauu add elements that are not set in u_parameter_set.yaml, 

108 ldaul also needs to be supplied for these elements. E.g., 

109 {"Ti": 2, "V": 2, "O": 1} 

110 ldaul_set_name: 

111 LDAUU and LDAUL default set name defined in u_parameter_set.yaml. 

112 npar_kpar (bool): 

113 Whether to automatically set the NPAR and KPAR tags. 

114 encut (float): 

115 Cutoff energy in eV. 

116 This argument is useful when performing a set of calculations 

117 for models with different elements yet their energies to be 

118 compared, e.g., defect calculations with impurity. 

119 is_magnetization (bool): 

120 Whether to have magnetization. If exists, set 

121 ISPIN = 2 and time_reversal = False, the latter of which is 

122 necessary for band structure calculations. 

123 num_nodes (int): 

124 Numbers of nodes used to determine KPAR and NPAR INCAR setting. 

125 structure_opt_encut_factor (float): 

126 ENCUT multiplied factor for structure optimization, where encut 

127 needs to be increased to reduce the Pulay Stress errors. 

128 """

193 def make_input(cls, 

194 structure: Structure, 

195 task: Union[str, Task] = "structure_opt", 

196 xc: Union[str, Xc] = "pbe", 

197 prev_set: "ViseInputSet" = None, 

198 abs_files_to_transfer: Optional[dict] = None, 

199 user_incar_settings: Optional[dict] = None, 

200 **kwargs) -> "ViseInputSet": 

201 """Construct ViseInputSet from some options including xc and task. 

202 

203 To make a simple but practical rule for inheriting the previous 

204 calculation condition and effectively use its results, we adopt 

205 fail-safe rule. 

206 

207 When, the prev_set is set, we compare task and xc between current 

208 input and prev_set, and inherit some options depending on whether 

209 task and/or xc are common. For instance, when we calculate the band 

210 structure with the HSE06 hybrid functional, it would be better to 

211 generate the WAVECAR file with e.g., PBE functional to reduce the 

212 computational cost. Then, the initial task was done with task=Task.band 

213 and xc=Xc.pbe with user_incar_setting={"LWAVE": True}, and next task is 

214 performed with task=Task.band, xc=Xc.hs. Then, task is common, so 

215 input set related to TaskStructureKpoints and TaskIncarSettings are 

216 inherited. Note that, CommonIncarSettings options are always inherited. 

217 

218 Other notes are as follows. 

219 

220 Note1: Charge set to structure is ignored when determining NELECT. For 

221 this purpose, charge option needs to be used. 

222 Note2: When the structure is changed via find_spglib_standard_primitive, 

223 all the site properties are removed. 

224 Note3: When different version of ViseInputSet with different defaults, 

225 the consistency of input set is destroyed, so the same version 

226 must be used when options are inherited. 

227 Note4: Other INCAR flags than those defined by TaskIncarSettings, 

228 XcIncarSettings, XcTaskIncarSettings, and CommonIncarSettings are 

229 not inherited. When some of them need to be inherited, they 

230 should be added to COMMON_OPTIONAL_FLAGS. 

231 Note5: user_incar_settings is not inherited from prev_set. One needs 

232 to explicitly specify it, again, if needed. 

233 Note6: Since default args are used in main function, set 

234 "structure_opt" and "pbe" strings as defaults. 

235 

236 Args: 

237 structure (Structure): 

238 The Structure to create inputs for. 

239 task (str/Task): 

240 Task defined in Task. 

241 xc (str/Xc): 

242 Exchange-correlation (xc) defined in Xc. 

243 prev_set (ViseInputSet): 

244 Previous ViseInputSet. 

245 abs_files_to_transfer (dict): 

246 Keys are file names with absolute paths to be transferred. 

247 Values mean the transfer modes, where there are three modes, 

248 "c": copy file, "m": move file, "l": make symbolic link 

249 e.g., {"/..../CHGCAR": "c", "/..../WAVECAR": "l", ..} 

250 user_incar_settings (dict): 

251 User INCAR settings. 

252 e.g., {"NSW": 100, "LWAVE": True, "LASPH": "A", ..} 

253 kwargs (dict): 

254 OPTION arguments. 

255 

256 Return: 

257 ViseInputSet class object. 

258 """ 

259 abs_files_to_transfer = abs_files_to_transfer or {} 

260 user_incar_settings = user_incar_settings or {} 

261 

262 if isinstance(task, str): 

263 task = Task.from_string(task) 

264 if isinstance(xc, str): 

265 xc = Xc.from_string(xc) 

266 

267 # First, set default. 

268 opts = deepcopy(cls.ALL_OPTIONS) 

269 

270 # Second, override with previous condition 

271 if prev_set: 

272 key_set = set(cls.COMMON_OPTIONS.keys()) 

273 if prev_set.task == task: 

274 key_set.update(cls.TASK_OPTIONS.keys()) 

275 if prev_set.xc == xc: 

276 key_set.update(cls.XC_OPTIONS.keys()) 

277 if prev_set.task == task and prev_set.xc == xc: 

278 key_set.update(cls.XC_TASK_OPTIONS) 

279 

280 for k in key_set: 

281 opts[k] = prev_set.kwargs[k] 

282 

283 # Third, override with keyword arguments 

284 for k in kwargs: 

285 if k not in opts.keys(): 

286 logger.warning(f"Keyword {k} is not adequate for ViseInputSet.") 

287 opts.update(kwargs) 

288 

289 task_str_kpt = TaskStructureKpoints.from_options( 

290 task=task, 

291 original_structure=structure, 

292 standardize_structure=opts["standardize_structure"], 

293 sort_structure=opts["sort_structure"], 

294 is_magnetization=opts["is_magnetization"], 

295 kpt_mode=opts["kpt_mode"], 

296 kpt_density=opts["kpt_density"], 

297 kpt_shift=opts["kpt_shift"], 

298 only_even=opts["only_even"], 

299 band_ref_dist=opts["band_ref_dist"], 

300 factor=opts["factor"], 

301 symprec=opts["symprec"], 

302 angle_tolerance=opts["angle_tolerance"]) 

303 

304 orig_matrix = structure.lattice.matrix 

305 matrix = task_str_kpt.structure.lattice.matrix 

306 structure_changed = \ 

307 not np.allclose(orig_matrix, matrix, atol=opts["symprec"]) 

308 

309 if structure_changed and opts["charge"] != 0: 

310 raise ValueError("Structure is changed but charge is set.") 

311 

312 if structure_changed: 

313 # The following files are useless when lattice is changed. 

314 pattern = \ 

315 re.compile("|".join([r"CHGCAR$", r"WAVECAR$", r"WAVEDER"])) 

316 for f in abs_files_to_transfer: 

317 if pattern.match(f): 

318 abs_files_to_transfer.pop(f, None) 

319 

320 # unique_justseen https://docs.python.org/ja/3/library/itertools.html 

321 # ["H", "H", "O", "O", "H"] -> ['H', 'O', 'H'] 

322 symbol_list = get_symbol_list(task_str_kpt.structure) 

323 

324 xc_task_potcar = XcTaskPotcar.from_options( 

325 xc=xc, 

326 symbol_list=symbol_list, 

327 potcar_set_name=opts["potcar_set_name"], 

328 override_potcar_set=opts["override_potcar_set"]) 

329 

330 task_settings = TaskIncarSettings.from_options( 

331 task=task, 

332 structure=task_str_kpt.structure, 

333 potcar=xc_task_potcar.potcar, 

334 num_kpoints=task_str_kpt.num_kpts, 

335 max_enmax=xc_task_potcar.max_enmax, 

336 is_magnetization=opts["is_magnetization"], 

337 vbm_cbm=opts["vbm_cbm"], 

338 npar_kpar=opts["npar_kpar"], 

339 num_nodes=opts["num_nodes"], 

340 encut=opts["encut"], 

341 structure_opt_encut_factor=opts["structure_opt_encut_factor"], 

342 dos_step_size=opts["dos_step_size"]) 

343 

344 xc_settings = XcIncarSettings.from_options( 

345 xc=xc, 

346 symbol_list=symbol_list, 

347 factor=task_str_kpt.factor, 

348 aexx=opts["aexx"], 

349 hubbard_u=opts["hubbard_u"], 

350 ldauu=opts["ldauu"], 

351 ldaul=opts["ldaul"], 

352 ldaul_set_name=opts["ldaul_set_name"]) 

353 

354 xc_task_settings = XcTaskIncarSettings.from_options() 

355 

356 common_settings = CommonIncarSettings.from_options( 

357 potcar=xc_task_potcar.potcar, 

358 composition=task_str_kpt.structure.composition, 

359 charge=opts["charge"]) 

360 

361 incar_settings = \ 

362 {**task_settings.settings, **xc_settings.settings, 

363 **xc_task_settings.settings, **common_settings.settings} 

364 

365 # user_incar_settings is the top prioritized. 

366 incar_settings.update(user_incar_settings) 

367 

368 # TODO: tweak the unfavorable combination of the input set. 

369 # e.g., Avoiding ICHARG = 11 is a must for hybrid functional. 

370 

371 return cls(structure=task_str_kpt.structure, 

372 task=task, 

373 xc=xc, 

374 kpoints=task_str_kpt.kpoints, 

375 potcar=xc_task_potcar.potcar, 

376 incar_settings=incar_settings, 

377 files_to_transfer=abs_files_to_transfer, **opts)

496 @classmethod 

497 def from_prev_calc(cls, 

498 dirname: str, 

499 task: Optional[Task] = None, 

500 xc: Optional[Xc] = None,

504 sort_structure: bool = True, 

505 standardize_structure: bool = False, 

506 files_to_transfer: Optional[dict] = None, 

507 user_incar_settings: Optional[dict] = None, 

508 contcar_filename: str = "CONTCAR", 

509 **kwargs) -> "ViseInputSet": 

510 """Constructor based on the previous calculations. 

511 

512 If the task and/or xc are not set, the previous ones are assumed. 

513 

514 Args: 

515 dirname (str): 

516 Directory name to be parsed. 

517 xc (Xc): 

518 Exchange-correlation (xc) defined in Xc. 

519 task (Task): 

520 Task defined in Task. 

521 json_filename (str): 

522 Json filename including ViseInputSet data. 

523 parse_calc_results (bool): 

524 Whether to parse the previous calculation results for band 

525 edges, magnetization. 

526 parse_incar (bool): 

527 Whether to parse the previous INCAR files, which could include 

528 other options that are not determined in vise. 

529 sort_structure (bool): 

530 Whether to sort the elements using get_sorted_structure method 

531 of Structure class in pymatgen. 

532 standardize_structure (bool): 

533 Whether to convert the structure to a standardized primitive. 

534 files_to_transfer (dict): 

535 Keys are file names to be transferred in the directory. 

536 Values mean the transfer modes, where there are three modes, 

537 "c": copy file, "m": move file, "l": make symbolic link 

538 e.g., {"CHGCAR": "c", "WAVECAR": "l", ..} 

539 user_incar_settings (dict): 

540 User INCAR settings. 

541 e.g., {"NSW": 100, "LWAVE": True, "LASPH": "A", ..} 

542 contcar_filename (str): 

543 CONTCAR type filename. Used if the user change the name. 

544 kwargs: 

545 Other OPTION arguments. 

546 

547 Return: 

548 ViseInputSet class object. 

549 """ 

550 kwargs = kwargs or {} 

551 kwargs["sort_structure"] = sort_structure 

552 kwargs["standardize_structure"] = standardize_structure 

553 

554 path = Path(dirname) 

555 

556 input_set = ViseInputSet.load_json(path / json_filename) 

557 

558 if parse_calc_results: 

559 vasprun, outcar = get_vasprun_outcar(dirname) 

560 structure = get_structure_from_prev_run(vasprun, outcar) 

561 

562 gap_properties = band_gap_properties(vasprun, outcar) 

563 if gap_properties: 

564 _, vbm, cbm = gap_properties 

565 kwargs["vbm_cbm"] = [vbm["energy"], cbm["energy"]] 

566 

567 if "magmom" in structure.site_properties.keys(): 

568 abs_max_mag = abs(max(structure.site_properties["magmom"], 

569 key=abs)) 

570 else: 

571 abs_max_mag = 0 

572 if vasprun.is_spin and abs_max_mag > 0.05: 

573 kwargs["is_magnetization"] = True 

574 

575 else: 

576 contcar = path / contcar_filename 

577 poscar = path / "POSCAR" 

578 if contcar.is_file and os.stat(contcar).st_size != 0: 

579 logger.info(f"{contcar} is parsed for structure.") 

580 structure = Structure.from_file(contcar) 

581 elif poscar.is_file and os.stat(poscar).st_size != 0: 

582 logger.warning(f"{poscar} is parsed for structure.") 

583 structure = Structure.from_file(poscar) 

584 else: 

585 raise FileNotFoundError("CONTCAR or POSCAR does not exist.") 

586 

587 if parse_incar: 

588 input_set.incar_settings = ViseIncar.from_file(path / "INCAR") 

589 

590 if files_to_transfer: 

591 abs_files_to_transfer = {} 

592 for filename, value in files_to_transfer.items(): 

593 f = path / filename 

594 if not f.is_file(): 

595 logger.warning(f"{f} does not exist.") 

596 elif os.stat(f).st_size == 0: 

597 logger.warning(f"{f} is empty.") 

598 elif value[0].lower() not in ["l", "c", "m"]: 

599 logger.warning(f"{value} option for {filename} is invalid.") 

600 else: 

601 abs_files_to_transfer[str(f)] = value[0].lower() 

602 files_to_transfer = abs_files_to_transfer 

603 

604 return cls.make_input(structure=structure,
